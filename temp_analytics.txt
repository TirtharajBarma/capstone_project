import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { useUser } from '@clerk/clerk-react';
import DashboardLayout from '../components/layout/DashboardLayout';
import { BarChart, Bar, XAxis, ResponsiveContainer, Tooltip, AreaChart, Area } from 'recharts';
import { historyAPI, handleAPIError } from '../services/api';
import { Skeleton, CardSkeleton, ChartSkeleton, TableSkeleton, StatsGridSkeleton } from '../components/ui';

const AnalyticsDashboard = () => {
  const { user } = useUser();
  const navigate = useNavigate();
  const [dateRange, setDateRange] = useState('30');
  const [stats, setStats] = useState({
    totalSubmissions: 0,
    accuracy: 0,
    accuracyChange: 0,
    mostCommonBreed: '',
    breedPercentage: 0,
    submissionsChange: 0
  });
  
  const [breedData, setBreedData] = useState([]);
  const [accuracyData, setAccuracyData] = useState([]);
  const [recentSubmissions, setRecentSubmissions] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchAnalytics = async () => {
      if (!user?.id) return;

      try {
        setLoading(true);

        const endDate = new Date();
        const startDate = new Date();
        startDate.setDate(startDate.getDate() - parseInt(dateRange));

        const statsResponse = await historyAPI.getStats({ 
          startDate: startDate.toISOString(),
          endDate: endDate.toISOString()
        });
        
        if (statsResponse.success) {
          const data = statsResponse.data;
          
          setStats({
            totalSubmissions: data.totalPredictions || 0,
            accuracy: data.avgConfidence || 0,
            accuracyChange: 0,
            mostCommonBreed: data.topBreeds?.[0]?._id || 'N/A',
            breedPercentage: data.topBreeds?.[0] ? Math.round((data.topBreeds[0].count / data.totalPredictions) * 100) : 0,
            submissionsChange: 0
          });

          if (data.topBreeds && data.topBreeds.length > 0) {
            setBreedData(data.topBreeds.slice(0, 7).map(breed => ({
              name: breed._id.substring(0, 10),
              value: breed.count
            })));
          }

          if (data.dailyAccuracy && data.dailyAccuracy.length > 0) {
            setAccuracyData(data.dailyAccuracy.map(d => ({
              name: d.date,
              accuracy: d.accuracy
            })));
          } else {
            setAccuracyData([]);
          }
        }

        const historyResponse = await historyAPI.getHistory({ 
          page: 1, 
          limit: 4 
        });

        if (historyResponse.success) {
          const predictions = historyResponse.data.predictions || [];
          setRecentSubmissions(predictions.map(pred => ({
            id: pred._id,
            breed: pred.predictedBreed,
            confidence: `${pred.confidencePercentage}%`,
            status: 'Successful',
            date: new Date(pred.createdAt).toISOString().split('T')[0],
            image: pred.imageUrl || `https://via.placeholder.com/400x300?text=${encodeURIComponent(pred.predictedBreed)}`
          })));
        }

      } catch (error) {
        console.error('Failed to fetch analytics:', error);
        const errorInfo = handleAPIError(error);
        console.error(errorInfo.message);
      } finally {
        setLoading(false);
      }
    };

    fetchAnalytics();
  }, [user?.id, dateRange]);

  if (loading) {
    return (
      <DashboardLayout>
        <div className="flex-1 p-4 lg:p-8">
          <div className="mx-auto flex w-full max-w-6xl flex-col gap-6">
            <div className="flex items-center justify-between">
              <Skeleton className="w-48 h-10" />
              <Skeleton className="w-36 h-12 rounded-xl" />
            </div>

            <StatsGridSkeleton />
            
            <div className="grid grid-cols-1 gap-6 lg:grid-cols-2">
              <ChartSkeleton />
              <ChartSkeleton />
            </div>

            <Skeleton className="w-64 h-8" />
            <TableSkeleton rows={4} />
          </div>
        </div>
      </DashboardLayout>
    );
  }

  return (
    <DashboardLayout>
      <div className="flex-1 p-4 lg:p-8">
        <div className="mx-auto flex w-full max-w-6xl flex-col gap-8">
          {/* Page Heading */}
          <header className="flex flex-wrap items-center justify-between gap-4 py-2 border-b border-slate-100 pb-6">
            <div className="flex flex-col gap-1">
              <h1 className="text-3xl font-bold tracking-tight text-slate-900">
                Analytics
              </h1>
              <p className="text-slate-500 font-medium tracking-wide">
                Insights and statistics from your image analyses
              </p>
            </div>
            <select 
              value={dateRange}
              onChange={(e) => setDateRange(e.target.value)}
              className="px-5 py-2.5 rounded-xl bg-white border border-slate-200 text-slate-700 text-[14px] font-semibold hover:bg-slate-50 hover:border-slate-300 transition-all cursor-pointer shadow-sm focus:outline-none focus:ring-2 focus:ring-indigo-500/20 active:scale-[0.98]"
            >
              <option value="7">Last 7 Days</option>
              <option value="30">Last 30 Days</option>
              <option value="90">Last 90 Days</option>
              <option value="365">Last Year</option>
            </select>
          </header>

          {/* Stats Cards */}
          <section className="grid grid-cols-1 gap-5 sm:grid-cols-3">
            <div className="rounded-2xl p-6 bg-white border border-slate-100 shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)] hover:shadow-md transition-shadow group relative overflow-hidden">
              <div className="absolute top-0 right-0 -mr-4 -mt-4 size-20 rounded-full bg-slate-50 group-hover:bg-indigo-50 transition-colors opacity-50"></div>
              <p className="text-[12px] font-semibold text-slate-500 tracking-wider uppercase mb-1 relative z-10">Total Predictions</p>
              <p className="mt-1 text-[2.5rem] font-bold text-slate-900 tracking-tighter relative z-10 leading-none">
                {stats.totalSubmissions.toLocaleString()}
              </p>
              {stats.submissionsChange !== 0 && (
                <div className="mt-4 flex items-center gap-1.5 relative z-10">
                  <span className={`flex size-5 items-center justify-center rounded-full ${stats.submissionsChange > 0 ? 'bg-emerald-100 text-emerald-700' : 'bg-red-100 text-red-700'}`}>
                    <span className="material-symbols-outlined text-[14px]">{stats.submissionsChange > 0 ? 'trending_up' : 'trending_down'}</span>
                  </span>
                  <p className={`text-[13px] font-semibold ${stats.submissionsChange > 0 ? 'text-emerald-700' : 'text-red-700'}`}>
                    {Math.abs(stats.submissionsChange)}% <span className="text-slate-400 font-medium">vs last period</span>
                  </p>
                </div>
              )}
            </div>
            <div className="rounded-2xl p-6 bg-white border border-slate-100 shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)] hover:shadow-md transition-shadow group relative overflow-hidden">
              <div className="absolute top-0 right-0 -mr-4 -mt-4 size-20 rounded-full bg-slate-50 group-hover:bg-amber-50 transition-colors opacity-50"></div>
              <p className="text-[12px] font-semibold text-slate-500 tracking-wider uppercase mb-1 relative z-10">Avg. Confidence</p>
              <p className="mt-1 text-[2.5rem] font-bold text-slate-900 tracking-tighter relative z-10 leading-none">
                {stats.accuracy}%
              </p>
              {stats.accuracyChange !== 0 && (
                <div className="mt-4 flex items-center gap-1.5 relative z-10">
                  <span className={`flex size-5 items-center justify-center rounded-full ${stats.accuracyChange > 0 ? 'bg-emerald-100 text-emerald-700' : 'bg-red-100 text-red-700'}`}>
                    <span className="material-symbols-outlined text-[14px]">{stats.accuracyChange > 0 ? 'trending_up' : 'trending_down'}</span>
                  </span>
                  <p className={`text-[13px] font-semibold ${stats.accuracyChange > 0 ? 'text-emerald-700' : 'text-red-700'}`}>
                    {Math.abs(stats.accuracyChange)}% <span className="text-slate-400 font-medium">vs last period</span>
                  </p>
                </div>
              )}
            </div>
            <div className="rounded-2xl p-6 bg-white border border-slate-100 shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)] hover:shadow-md transition-shadow group relative overflow-hidden">
              <div className="absolute top-0 right-0 -mr-4 -mt-4 size-20 rounded-full bg-slate-50 group-hover:bg-emerald-50 transition-colors opacity-50"></div>
              <p className="text-[12px] font-semibold text-slate-500 tracking-wider uppercase mb-2 relative z-10">Top Breed</p>
              <p className="mt-1 text-2xl font-bold text-slate-900 tracking-tight truncate relative z-10">
                {stats.mostCommonBreed || 'N/A'}
              </p>
              <div className="mt-2.5 flex items-center gap-2 relative z-10">
                <div className="h-1.5 w-full bg-slate-100 rounded-full overflow-hidden">
                  <div className="h-full bg-emerald-500 rounded-full" style={{ width: `${stats.breedPercentage}%` }}></div>
                </div>
                <p className="text-[13px] font-medium text-slate-500 whitespace-nowrap">
                  <span className="text-slate-800 font-bold">{stats.breedPercentage}%</span> of total
                </p>
              </div>
            </div>
          </section>

          {/* Charts */}
          <section className="grid grid-cols-1 gap-6 lg:grid-cols-2">
            <div className="rounded-2xl border border-slate-100 p-6 bg-white shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)]">
              <div className="mb-6 flex flex-col gap-1">
                <p className="text-lg font-bold tracking-tight text-slate-900">Breed Distribution</p>
                <p className="text-[13px] font-medium text-slate-500">Top 7 breeds in selected period</p>
              </div>
              <div style={{ width: '100%', height: 280 }}>
                {breedData.length > 0 ? (
                  <ResponsiveContainer>
                    <BarChart data={breedData} margin={{ top: 10, right: 10, left: -20, bottom: 0 }}>
                      <XAxis 
                        dataKey="name" 
                        axisLine={false} 
                        tickLine={false} 
                        tick={{ fill: '#64748b', fontSize: 12, fontWeight: 500 }} 
                        dy={10} 
                      />
                      <Tooltip 
                        cursor={{ fill: '#f8fafc' }}
                        contentStyle={{ borderRadius: '12px', border: 'none', boxShadow: '0 4px 20px -5px rgba(0,0,0,0.1)', fontWeight: 500, color: '#0f172a' }}
                      />
                      <Bar 
                        dataKey="value" 
                        fill="#4f46e5" 
                        radius={[6, 6, 6, 6]} 
                        barSize={32}
                        animationDuration={1500}
                        animationEasing="ease-out"
                      />
                    </BarChart>
                  </ResponsiveContainer>
                ) : (
                  <div className="flex h-full items-center justify-center border-2 border-dashed border-slate-100 rounded-xl bg-slate-50/50">
                    <p className="text-slate-500 font-medium">No data available for this period</p>
                  </div>
                )}
              </div>
            </div>

            <div className="rounded-2xl border border-slate-100 p-6 bg-white shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)]">
              <div className="mb-6 flex flex-col gap-1">
                <p className="text-lg font-bold tracking-tight text-slate-900">Accuracy Trend</p>
                <p className="text-[13px] font-medium text-slate-500">Average confidence over time</p>
              </div>
              <div style={{ width: '100%', height: 280 }}>
                {accuracyData.length > 0 ? (
                  <ResponsiveContainer>
                    <AreaChart data={accuracyData} margin={{ top: 10, right: 10, left: -20, bottom: 0 }}>
                      <defs>
                        <linearGradient id="colorAccuracy" x1="0" y1="0" x2="0" y2="1">
                          <stop offset="5%" stopColor="#10b981" stopOpacity={0.2}/>
                          <stop offset="95%" stopColor="#10b981" stopOpacity={0}/>
                        </linearGradient>
                      </defs>
                      <XAxis 
                        dataKey="name" 
                        axisLine={false} 
                        tickLine={false} 
                        tick={{ fill: '#64748b', fontSize: 12, fontWeight: 500 }}
                        dy={10}
                      />
                      <Tooltip 
                        contentStyle={{ borderRadius: '12px', border: 'none', boxShadow: '0 4px 20px -5px rgba(0,0,0,0.1)', fontWeight: 500 }}
                        itemStyle={{ color: '#0f172a' }}
                      />
                      <Area 
                        type="monotone" 
                        dataKey="accuracy" 
                        stroke="#10b981" 
                        strokeWidth={3}
                        fillOpacity={1} 
                        fill="url(#colorAccuracy)" 
                        animationDuration={1500}
                        animationEasing="ease-out"
                        activeDot={{ r: 6, strokeWidth: 0, fill: '#10b981' }}
                      />
                    </AreaChart>
                  </ResponsiveContainer>
                ) : (
                  <div className="flex h-full items-center justify-center border-2 border-dashed border-slate-100 rounded-xl bg-slate-50/50">
                    <p className="text-slate-500 font-medium">No data available for this period</p>
                  </div>
                )}
              </div>
            </div>
          </section>

          {/* Recent Submissions */}
          <section>
            <h2 className="text-xl font-bold tracking-tight text-slate-900 mb-5">Recent Predictions</h2>
            
            {recentSubmissions.length === 0 ? (
              <div className="flex flex-col items-center justify-center gap-3 rounded-2xl border-2 border-dashed border-slate-200 bg-white p-12 text-center shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)]">
                <div className="w-16 h-16 rounded-full bg-slate-50 flex items-center justify-center ring-1 ring-slate-100">
                  <svg className="w-8 h-8 text-slate-300" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={1.5} d="M9 17v-2m3 2v-4m3 4v-6m2 10H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z" />
                  </svg>
                </div>
                <p className="text-slate-500 font-medium">No predictions yet. Start by making your first prediction!</p>
              </div>
            ) : (
              <div className="overflow-x-auto rounded-3xl border border-slate-100 bg-white shadow-[0_2px_10px_-4px_rgba(0,0,0,0.05)]">
                <table className="w-full text-left whitespace-nowrap">
                  <thead className="bg-slate-50">
                    <tr>
                      <th className="px-6 py-4 text-xs font-semibold text-slate-500 uppercase tracking-wider">Image</th>
                      <th className="px-6 py-4 text-xs font-semibold text-slate-500 uppercase tracking-wider">Breed</th>
                      <th className="px-6 py-4 text-xs font-semibold text-slate-500 uppercase tracking-wider">Confidence</th>
                      <th className="px-6 py-4 text-xs font-semibold text-slate-500 uppercase tracking-wider">Date</th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-slate-100">
                    {recentSubmissions.map((submission) => (
                      <tr 
                        key={submission.id} 
                        className="hover:bg-slate-50/50 cursor-pointer transition-colors group"
                        onClick={() => {
                          const predictionData = {
                            breed: submission.breed,
                            confidence: parseFloat(submission.confidence) / 100,
                          };
                          navigate('/results', {
                            state: {
                              prediction: predictionData,
                              imageUrl: submission.image,
                              saved: true
                            }
                          });
                        }}
                      >
                        <td className="p-4 px-6">
                          <div 
                            className="w-14 h-14 rounded-xl bg-cover bg-center ring-1 ring-slate-900/5 group-hover:scale-105 transition-transform"
                            style={{ backgroundImage: `url('${submission.image}')` }}
                          />
                        </td>
                        <td className="px-6 py-4 font-bold tracking-tight text-slate-900 group-hover:text-indigo-600 transition-colors">{submission.breed}</td>
                        <td className="px-6 py-4">
                          <span className="inline-flex items-center px-2.5 py-1 rounded-md text-[13px] font-semibold bg-emerald-50 text-emerald-700">
                            {submission.confidence}
                          </span>
                        </td>
                        <td className="px-6 py-4 text-slate-500 font-medium text-[13px]">{submission.date}</td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
            )}
          </section>
        </div>
      </div>
    </DashboardLayout>
  );
};

export default AnalyticsDashboard;
